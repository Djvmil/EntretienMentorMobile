 @Composable
 fun ShimmerCircle(
     isEnableShimmer: Boolean,
     isLightModeActive: Boolean = false,
     size: Dp,
     padding: Dp
 ) {
     Box(
         modifier = Modifier
             .padding(all = padding)
             .background(
                 color = Color.LightGray,
                 shape = CircleShape
             )
             .size(size)
             .shimmerLoadingAnimation(
                 isEnableShimmer = isEnableShimmer,
                 isLightModeActive = isLightModeActive,
             )
     )
 }
